 function rains(x,y)
	 local r={
		 x=flr(rnd(162)),
		 y=-8,
  }
  add(rain,r)
 end
 
 function drops(x,y)
	 local d={
		 x=x,
		 y=y,
		 yy=y,
	  rndx=-1+flr(rnd(3)),
	  rndy=1+rnd(2),
  }
  add(drop,d)
 end
 
 function seeds(x,y)
	 local se={
		 x=x,
		 y=y,
		 yy=flr(rnd(5))+4,
	  rndx=(flr(rnd(3))+1),
	  rndy=(rnd(3)+1)/10,
	  active=1,
  }
  add(seed,se)
 end

 function waters(x,y)
	 local wat={
		 x=x,
		 y=y,
	  rndx=(flr(rnd(3))+1),
	  rndy=.5+rnd(1),
	  watered=0,
  }
  add(water,wat)
 end
 
 function eggs(x,y)

 end

 function hcenter(s)
